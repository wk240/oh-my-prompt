   ```typescript
   import { ScrollArea } from './ui/scroll-area'
   import { cn } from '../../lib/utils'

   function CategorySidebar() {
     const { categories, selectedCategoryId, setSelectedCategory } = usePromptStore()
     
     return (
       <div className="w-[80px] h-full border-r border-border">
         <ScrollArea className="h-full">
           {categories.map(category => (
             <button
               key={category.id}
               onClick={() => setSelectedCategory(category.id)}
               className={cn(
                 "w-full px-2 py-3 text-sm text-left hover:bg-muted transition-colors",
                 selectedCategoryId === category.id && "bg-primary/10 text-primary font-medium"
               )}
             >
               {category.name}
             </button>
           ))}
         </ScrollArea>
       </div>
     )
   }
   ```
