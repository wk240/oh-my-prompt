1. Remove sample data imports, add storage retrieval:
   ```typescript
   // Remove:
   // import { SAMPLE_PROMPTS, SAMPLE_CATEGORIES } from '../sample-data'

   import { MessageType } from '../../shared/messages'
   import type { StorageSchema, Prompt, Category } from '../../shared/types'
   import { useState, useEffect } from 'react'
   ```

2. Add storage state:
   ```typescript
   function DropdownApp() {
     const [prompts, setPrompts] = useState<Prompt[]>([])
     const [categories, setCategories] = useState<Category[]>([])
     const [isLoading, setIsLoading] = useState(true)
     
     useEffect(() => {
       // Fetch storage data on mount
       chrome.runtime.sendMessage(
         { type: MessageType.GET_STORAGE },
         (response) => {
           if (response?.success && response.data) {
             const data = response.data as StorageSchema
             setPrompts(data.prompts)
             setCategories(data.categories)
           }
           setIsLoading(false)
         }
       )
     }, [])
     
     // ... rest of component using prompts/categories state
   }
   ```

3. Update getPromptsByCategory to use state:
   ```typescript
   const getPromptsByCategory = (categoryId: string) => {
     return prompts.filter(p => p.categoryId === categoryId)
   }
   ```

4. Add loading state UI:
   ```typescript
   if (isLoading) {
     return (
       <div className="p-4 text-center text-sm text-gray-500">
         加载中...
       </div>
     )
   }
   ```

5. Handle empty state (no prompts):
   ```typescript
   if (prompts.length === 0) {
     return (
       <div className="p-4 text-center text-sm text-gray-500">
         暂无提示词，请在插件中添加
       </div>
     )
   }
   ```

6. Keep existing category selection and prompt insertion logic unchanged.
